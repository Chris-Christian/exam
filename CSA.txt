//add 8 bit
DATA SEGMENT
X DB 3BH
Y DB 6AH
DATA ENDS

CODE SEGMENT
ASSUME CS: CODE , DS: DATA
START: MOV AX,DATA
       MOV DS,AX
       MOV AH,X
       MOV AL,Y
       ADD AH,AL
       INT 3H

CODE ENDS
END START


//multiply
DATA SEGMENT
X DB 5H
Y DB 6H
DATA ENDS

CODE SEGMENT
ASSUME CS: CODE , DS: DATA
START: MOV AX,DATA
       MOV DS,AX
       MOV AH,X
       MOV BL,Y
       MUL BL
       INT 3H

CODE ENDS
END START

//Q1. Write an assembly language program to multiply two 16-bit words.

DATA SEGMENT
NUM1 DW 1234H ; First 16-bit number
NUM2 DW 5678H ; Second 16-bit number
RESULT DW ?
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START: MOV AX, DATA ; Load address of DATA segment into AX
  MOV DS, AX
  MOV AX, NUM1
  MOV BX, NUM2
  MUL BX
  MOV RESULT, AX
  INT 3H ; Terminate program
CODE ENDS
END START

  HELLO DB "HELLO MY NAME IS CHRIS", "$"
DATA ENDS
CODE SEGMENT
  ASSUME CS:CODE, DS:DATA
  START:
    MOV AX,DATA
    MOV DS,AX
  LEA DX,HELLO
  MOV AH,09H
  INT 21H
  STOP:
    MOV AX,4C00H
    INT 21H
CODE ENDS
END START
